Of course, you can (and should) implement your own to_json() method. But if you want a reference and guidance, here it is:

You need to make one small minor change to the FHtree.h header file originally written by Professor Loceff. Add the following lines to the bottom of the FHtree class definition:

    // New in V2
    public:
        string to_json() const { return to_json(mRoot, 0); }
    protected:
        string to_json(FHtreeNode<Object> *treeNode, int level) const;

Then add the implementation of to_json() out of line, below the class definition: 

    // New in V2
    template <class Object>
    string FHtree<Object>::to_json(FHtreeNode<Object> *treeNode, int level) const {
        static string blankString = "                                    ";
        string indent = blankString.substr(0, level);

        string result = indent + "{\n";

        result += indent + " \"name\": \"" + treeNode->data + "\"";
        result += treeNode->firstChild? ",\n" : "\n";

        // Omit serializing non-existing children.
        if (treeNode->firstChild) {
            result += indent + " \"children\": [\n";

            for (FHtreeNode<Object> *p = treeNode->firstChild; p != NULL; p = p->sib) {
                if (p != treeNode->firstChild) result += ",\n";
                result += to_json(p, level+2);
            }

            result += "\n" + indent + " ]\n";
        }

        result += indent + "}";
        return result;
    }

And that's it. You can now call the JSON serializer on your FHtrees like so: 

    std::cout <<  myTree.to_json() <<endl;
